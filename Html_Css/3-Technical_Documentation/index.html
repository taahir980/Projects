int BinarySearch(int arr[], int low, int high, int trgt) { // Base Case if(low > high) return -1; int mid = (low + high) / 2; if(arr[mid] == trgt) return mid; else if(trgt > arr[mid]) BinarySearch(arr, mid+1, high, trgt); else BinarySearch(arr, low, mid-1, trgt); }

void selection_sort(int arr[], int n) { for(int i=0; i=n-2; i++) { int miniIdx = i; for(int j=i+1; j=n-1; j++) { if(arr[j] arr[miniIdx]) miniIdx = j; // T.C = O(n^2) } swap(arr[miniIdx], arr[i]); } }
